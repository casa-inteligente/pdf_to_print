import pandas as pd

# Suponha que 'pdf' é o seu arquivo PDF e 'tables' são as tabelas extraídas
tables = pdf.tables

# Crie um DataFrame vazio para armazenar todas as tabelas
all_tables = pd.DataFrame()

# Loop através de todas as tabelas e adicione-as ao DataFrame
for table in tables:
    df = table.df  # Converta a tabela em um DataFrame
    all_tables = pd.concat([all_tables, df])  # Adicione o DataFrame à tabela principal

# Agora 'all_tables' contém todas as suas tabelas combinadas
